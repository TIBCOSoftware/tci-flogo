    94	func blockGeneric(dig *digest, p []byte) {
    95		var w [80]uint64
    96		h0, h1, h2, h3, h4, h5, h6, h7 := dig.h[0], dig.h[1], dig.h[2], dig.h[3], dig.h[4], dig.h[5], dig.h[6], dig.h[7]
    97		for len(p) >= chunk {
    98			for i := 0; i < 16; i++ {
    99				j := i * 8
   100				w[i] = uint64(p[j])<<56 | uint64(p[j+1])<<48 | uint64(p[j+2])<<40 | uint64(p[j+3])<<32 |
   101					uint64(p[j+4])<<24 | uint64(p[j+5])<<16 | uint64(p[j+6])<<8 | uint64(p[j+7])
   102			}
   103			for i := 16; i < 80; i++ {
   104				v1 := w[i-2]
   105				t1 := (v1>>19 | v1<<(64-19)) ^ (v1>>61 | v1<<(64-61)) ^ (v1 >> 6)
   106				v2 := w[i-15]
   107				t2 := (v2>>1 | v2<<(64-1)) ^ (v2>>8 | v2<<(64-8)) ^ (v2 >> 7)
   108	
   109				w[i] = t1 + w[i-7] + t2 + w[i-16]
   110			}
   111	
   112			a, b, c, d, e, f, g, h := h0, h1, h2, h3, h4, h5, h6, h7
   113	
   114			for i := 0; i < 80; i++ {
   115				t1 := h + ((e>>14 | e<<(64-14)) ^ (e>>18 | e<<(64-18)) ^ (e>>41 | e<<(64-41))) + ((e & f) ^ (^e & g)) + _K[i] + w[i]
   116	
   117				t2 := ((a>>28 | a<<(64-28)) ^ (a>>34 | a<<(64-34)) ^ (a>>39 | a<<(64-39))) + ((a & b) ^ (a & c) ^ (b & c))
   118	
   119				h = g
   120				g = f
   121				f = e
   122				e = d + t1
   123				d = c
   124				c = b
   125				b = a
   126				a = t1 + t2
   127			}
   128	
   129			h0 += a
   130			h1 += b
   131			h2 += c
   132			h3 += d
   133			h4 += e
   134			h5 += f
   135			h6 += g
   136			h7 += h
   137	
   138			p = p[chunk:]
   139		}
