  func Gettimeofday(tv *Timeval) (err error) {
  	errno := gettimeofday(tv)
  	if errno != 0 {
  		return errno
  	}
  	return nil
  }
  
  func Time(t *Time_t) (tt Time_t, err error) {
  	var tv Timeval
  	errno := gettimeofday(&tv)
  	if errno != 0 {
  		return 0, errno
  	}
  	if t != nil {
  		*t = Time_t(tv.Sec)
  	}
  	return Time_t(tv.Sec), nil
  }

  //sysnb	pipe(p *[2]_C_int) (err error)
  
  func Pipe(p []int) (err error) {
  	if len(p) != 2 {
  		return EINVAL
  	}
  	var pp [2]_C_int
  	err = pipe(&pp)
  	p[0] = int(pp[0])
  	p[1] = int(pp[1])
  	return
  }
  
  //sysnb pipe2(p *[2]_C_int, flags int) (err error)
  
  func Pipe2(p []int, flags int) (err error) {
  	if len(p) != 2 {
  		return EINVAL
  	}
  	var pp [2]_C_int
  	err = pipe2(&pp, flags)
  	p[0] = int(pp[0])
  	p[1] = int(pp[1])
  	return
  }
