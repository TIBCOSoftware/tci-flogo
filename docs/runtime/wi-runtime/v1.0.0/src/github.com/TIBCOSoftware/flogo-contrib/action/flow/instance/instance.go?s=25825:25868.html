   193	// DoStep performs a single execution 'step' of the Flow Instance
   194	func (pi *Instance) DoStep() bool {
   195	
   196		hasNext := false
   197	
   198		pi.ResetChanges()
   199	
   200		pi.stepID++
   201	
   202		if pi.status == StatusActive {
   203	
   204			item, ok := pi.WorkItemQueue.Pop()
   205	
   206			if ok {
   207				logger.Debug("popped item off queue")
   208	
   209				workItem := item.(*WorkItem)
   210	
   211				pi.ChangeTracker.trackWorkItem(&WorkItemQueueChange{ChgType: CtDel, ID: workItem.ID, WorkItem: workItem})
   212	
   213				pi.execTask(workItem)
   214				hasNext = true
   215			} else {
   216				logger.Debug("queue emtpy")
   217			}
   218		}
   219	
   220		return hasNext
   221	}

   239	func (pi *Instance) scheduleEval(taskData *TaskData, evalCode int) {
   240	
   241		pi.wiCounter++
   242	
   243		workItem := NewWorkItem(pi.wiCounter, taskData, EtEval, evalCode)
   244		logger.Debugf("Scheduling EVAL on task: %s\n", taskData.task.Name())
   245

   250	// execTask executes the specified Work Item of the Flow Instance
   251	func (pi *Instance) execTask(workItem *WorkItem) {
   252	
   253		defer func() {
   254			if r := recover(); r != nil {
   255	
   256				err := fmt.Errorf("Unhandled Error executing task '%s' : %v\n", workItem.TaskData.task.Name(), r)
   257				logger.Error(err)
   258	
   259				// todo: useful for debugging
   260				logger.Debugf("StackTrace: %s", debug.Stack())
   261	
   262				pi.appendActivityErrorData(workItem.TaskData, activity.NewError(err.Error(), "", nil))
   263				if workItem.TaskData.taskEnv.ID != idEhTasEnv {
   264					//not already in global handler, so handle it
   265					pi.HandleGlobalError()
   266				}
   267			}
   268		}()
   269	
   270		taskData := workItem.TaskData
   271		taskBehavior := pi.FlowModel.GetTaskBehavior(taskData.task.TypeID())
   272	
   273		var done bool
   274		var doneCode int
   275		var err error
   276	
   277		//todo: should validate process activities
   278	
   279		if workItem.ExecType == EtEval {
   280	
   281			eval := true
   282	
   283			if taskData.HasAttrs() {
   284	
   285				err := applyInputMapper(pi, taskData)
   286	
   287				if err != nil {
   288					pi.appendMapperErrorData(err)
   289					pi.HandleGlobalError()
   290					return
   291				}
   292	
   293				eval = applyInputInterceptor(pi, taskData)
   294			}
   295	
   296			if eval {
   297				done, doneCode, err = pi.evalTask(taskBehavior, taskData, workItem.EvalCode)
   298			} else {
   299				done = true
   300			}
   301		} else {
   302			done, doneCode, err = taskBehavior.PostEval(taskData, workItem.EvalCode, nil)
   303		}
   304	
   305		if err != nil {
   306			pi.handleTaskError(taskBehavior, taskData, err)
   307			return
   308		}
   309	
   310		if done {
   311	
   312			if taskData.HasAttrs() {
   313				applyOutputInterceptor(pi, taskData)
   314	
   315				appliedMapper, err := applyOutputMapper(pi, taskData)
   316	
   317				if err != nil {
   318					pi.appendMapperErrorData(err)
   319					pi.HandleGlobalError()
   320					return
   321				}
   322	
   323				if !appliedMapper && !taskData.task.IsScope() {
   324	
   325					logger.Debug("Applying Default Output Mapping")
   326					//applyDefaultActivityOutputMappings(pi, taskData)
   327				}
   328			}
   329	
   330			pi.handleTaskDone(taskBehavior, taskData, doneCode)
   331		}
   332	}

   334	func (pi *Instance) evalTask(taskBehavior model.TaskBehavior, taskData *TaskData, evalCode int) (done bool, doneCode int, err error) {
   335	
   336		defer func() {
   337			if r := recover(); r != nil {
   338	
   339				err = fmt.Errorf("Unhandled Error evaluating task '%s' : %v\n", taskData.task.Name(), r)
   340				logger.Error(err)
   341	
   342				// todo: useful for debugging
   343				logger.Debugf("StackTrace: %s", debug.Stack())
   344	
   345				done = false
   346				doneCode = 0
   347			}
   348		}()
   349	
   350		done, doneCode, err = taskBehavior.Eval(taskData, evalCode)
   351	
   352		return done, doneCode, err
   353	}

   355	// handleTaskDone handles the completion of a task in the Flow Instance
   356	func (pi *Instance) handleTaskDone(taskBehavior model.TaskBehavior, taskData *TaskData, doneCode int) {
   357	
   358		notifyParent, childDoneCode, taskEntries, err := taskBehavior.Done(taskData, doneCode)
   359	
   360		if err != nil {
   361			pi.appendErrorData(err)
   362			pi.HandleGlobalError()
   363			return
   364		}
   365	
   366		task := taskData.Task()
   367	
   368		if notifyParent {
   369	
   370			parentTask := task.Parent()
   371	
   372			if parentTask != nil {
   373				parentTaskData := taskData.taskEnv.TaskDatas[parentTask.ID()]
   374				parentBehavior := pi.FlowModel.GetTaskBehavior(parentTask.TypeID())
   375				parentDone, parentDoneCode := parentBehavior.ChildDone(parentTaskData, task, childDoneCode)
   376	
   377				if parentDone {
   378					pi.handleTaskDone(parentBehavior, parentTaskData, parentDoneCode)
   379				}
   380	
   381			} else {
   382	
   383				//todo distinguish between error handler env and rootTaskEnv
   384	
   385				//Root Task is Done, so notify Flow
   386				flowBehavior := pi.FlowModel.GetFlowBehavior()
   387				flowBehavior.TasksDone(pi, childDoneCode)
   388				flowBehavior.Done(pi)
   389	
   390				pi.setStatus(StatusCompleted)
   391			}
   392		}
   393	
   394		if len(taskEntries) > 0 {
   395	
   396			for _, taskEntry := range taskEntries {
   397	
   398				logger.Debugf("execTask - TaskEntry: %v\n", taskEntry)
   399				taskToEnterBehavior := pi.FlowModel.GetTaskBehavior(taskEntry.Task.TypeID())
   400	
   401				enterTaskData, _ := taskData.taskEnv.FindOrCreateTaskData(taskEntry.Task)
   402	
   403				eval, evalCode := taskToEnterBehavior.Enter(enterTaskData, taskEntry.EnterCode)
   404	
   405				if eval {
   406					pi.scheduleEval(enterTaskData, evalCode)
   407				}
   408			}
   409		}
   410	
   411		taskData.taskEnv.releaseTask(task)
   412	}

   444	// handleTaskError handles the completion of a task in the Flow Instance
   445	func (pi *Instance) handleTaskError(taskBehavior model.TaskBehavior, taskData *TaskData, err error) {
   446	
   447		handled, taskEntry := taskBehavior.Error(taskData)
   448	
   449		if !handled {
   450			pi.appendActivityErrorData(taskData, err)
   451			if taskData.taskEnv.ID != idEhTasEnv {
   452				//not already in global handler, so handle it
   453				pi.HandleGlobalError()
   454			}
   455			return
   456		}
   457	
   458		//todo add error data for task to flow
   459	
   460		if taskEntry != nil {
   461	
   462			logger.Debugf("execTask - TaskEntry: %v\n", taskEntry)
   463			taskToEnterBehavior := pi.FlowModel.GetTaskBehavior(taskEntry.Task.TypeID())
   464	
   465			enterTaskData, _ := taskData.taskEnv.FindOrCreateTaskData(taskEntry.Task)
   466	
   467			eval, evalCode := taskToEnterBehavior.Enter(enterTaskData, taskEntry.EnterCode)
   468	
   469			if eval {
   470				pi.scheduleEval(enterTaskData, evalCode)
   471			}
   472		}
   473	
   474		task := taskData.Task()
   475		taskData.taskEnv.releaseTask(task)
   476	}

   478	// HandleGlobalError handles instance errors
   479	func (pi *Instance) HandleGlobalError() {
   480	
   481		if pi.Flow.ErrorHandlerTask() != nil {
   482	
   483			ehTask := pi.Flow.ErrorHandlerTask()
   484	
   485			if pi.EhTaskEnv == nil {
   486				var taskEnv TaskEnv
   487				taskEnv.ID = idEhTasEnv
   488				taskEnv.Task = ehTask
   489				taskEnv.taskID = ehTask.ID()
   490				taskEnv.Instance = pi
   491				taskEnv.TaskDatas = make(map[int]*TaskData)
   492				taskEnv.LinkDatas = make(map[int]*LinkData)
   493	
   494				pi.EhTaskEnv = &taskEnv
   495			}
   496	
   497			ehTaskData := pi.EhTaskEnv.TaskDatas[ehTask.ID()]
   498	
   499			if ehTaskData == nil {
   500				ehTaskData = pi.EhTaskEnv.NewTaskData(ehTask)
   501			}
   502	
   503			//todo: should we clear out the existing workitem queue?
   504	
   505			pi.scheduleEval(ehTaskData, 0)
   506		} else {
   507	
   508			//todo: log error information
   509			pi.setStatus(StatusFailed)
   510		}
   511	}

   587	// init initializes the Task Environment, typically called on deserialization
   588	func (te *TaskEnv) init(flowInst *Instance) {

   590		if te.Instance == nil {

   594

   603			}
   604		}

   608	// specified task the task environment
   609	func (te *TaskEnv) FindOrCreateTaskData(task *definition.Task) (taskData *TaskData, created bool) {
   610	
   611		taskData, ok := te.TaskDatas[task.ID()]
   612	
   613		created = false
   614

   617			te.TaskDatas[task.ID()] = taskData
   618			te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtAdd, ID: task.ID(), TaskData: taskData})
   619	
   620			created = true
   621		}
   622	
   623		return taskData, created
   624	}

   653	// releaseTask cleans up TaskData in the task environment any of its dependencies.
   654	// This is called when a task is completed and can be discarded
   655	func (te *TaskEnv) releaseTask(task *definition.Task) {
   656		delete(te.TaskDatas, task.ID())
   657		te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtDel, ID: task.ID()})
   658	
   659		childTasks := task.ChildTasks()
   660	
   661		if len(childTasks) > 0 {
   662	
   663			for _, childTask := range childTasks {
   664				delete(te.TaskDatas, childTask.ID())
   665				te.Instance.ChangeTracker.trackTaskData(&TaskDataChange{ChgType: CtDel, ID: childTask.ID()})
   666			}
   667		}
   668	
   669		links := task.FromLinks()
   670	
   671		for _, link := range links {
   672			delete(te.LinkDatas, link.ID())
   673			te.Instance.ChangeTracker.trackLinkData(&LinkDataChange{ChgType: CtDel, ID: link.ID()})
   674		}
   675	}

   857	// EvalLink implements activity.ActivityContext.EvalLink method
   858	func (td *TaskData) EvalLink(link *definition.Link) (result bool, err error) {
   859	
   860		logger.Debugf("TaskContext.EvalLink: %d\n", link.ID())
   861	
   862		defer func() {
   863			if r := recover(); r != nil {
   864				logger.Warnf("Unhandled Error evaluating link '%s' : %v\n", link.ID(), r)
   865	
   866				// todo: useful for debugging
   867				logger.Debugf("StackTrace: %s", debug.Stack())
   868	
   869				if err != nil {
   870					err = fmt.Errorf("%v", r)
   871				}
   872			}
   873		}()
   874	
   875		mgr := td.taskEnv.Instance.Flow.GetLinkExprManager()
   876	
   877		if mgr != nil {
   878			result, err = mgr.EvalLinkExpr(link, td.taskEnv.Instance)
   879			return result, err
   880		}
   881	
   882		return true, nil
   883	}

   890	// EvalActivity implements activity.ActivityContext.EvalActivity method
   891	func (td *TaskData) EvalActivity() (done bool, evalErr error) {
   892	
   893		act := activity.Get(td.task.ActivityRef())
   894	
   895		//todo: if act == nil, return TaskDoesntHaveActivity error or something like that
   896	
   897		defer func() {
   898			if r := recover(); r != nil {
   899				logger.Warnf("Unhandled Error executing activity '%s'[%s] : %v\n", td.task.Name(), td.task.ActivityRef(), r)
   900	
   901				// todo: useful for debugging
   902				logger.Debugf("StackTrace: %s", debug.Stack())
   903	
   904				if evalErr == nil {
   905					evalErr = activity.NewError(fmt.Sprintf("%v", r), "", nil)
   906					done = false
   907				}
   908			}
   909			if evalErr != nil {
   910				logger.Errorf("Execution failed for Activity[%s] in Flow[%s] - %s", td.task.Name(), td.taskEnv.Instance.Name(), evalErr.Error())
   911			}
   912		}()
   913	
   914		done, evalErr = act.Eval(td)
   915	
   916		return done, evalErr
   917	}
